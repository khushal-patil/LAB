MACRO
INCR &X,&Y,&REG1 = AREG
MOVER &REG1,&X
ADD &REG1,&Y
MOVEM &REG1,&X
MEND
MACRO
DECR &A,&B,&REG2 = BREG
MOVER &REG2,&A
SUB &REG2,&B
MOVEM &REG2,&A
MEND
MACRO
KHUSH &P,&Q,&REG2 = BREG
MOVER &REG2,&P
SUB &REG2,&Q
MOVEM &REG2,&P
MEND
START 100
READ N1
READ N2
READ N3
READ N4
INCR N1,N2
DECR N1,N2
KHUSH N3,N4
STOP
N1 DS 1
N2 DS 2
N3 DS 1
N4 DS 1
END
